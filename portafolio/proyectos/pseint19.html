Solucion: 

                
                    Proceso miApp
                        Definir num Como Entero;
                        Definir control Como Entero;
                        
                        
                        Escribir "Ingrese un numero del 1 al 10";
                        Leer num;
                        Para control<-1 Hasta 10 Con Paso 1 Hacer
                            Escribir control," x ",num," = ", control*num;
                        FinPara
                        Escribir "¡Fin de programa!";
